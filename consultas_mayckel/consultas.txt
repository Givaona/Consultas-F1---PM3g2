Pergunta: (os donos da pista) quais pilotos tem as menores voltas de todos os tempos de acordo com as pistas?

Consulta:
SELECT M.RACE_ID, D.DRIVER_ID, D.FORENAME, D.SURNAME, D.NUMBER, M.NAME, M.TIME, M.TEMPO, M.YEAR
FROM(
	SELECT A.RACE_ID, A.DRIVER_ID, A.POSITION, A.TIME, MIN(A.MILISECONDS) AS TEMPO, B.CIRCUIT_ID, B.YEAR, B.NAME 
	FROM LAP_TIMES A
	INNER JOIN RACES B
	ON B.RACE_ID = A.RACE_ID
	GROUP BY B.CIRCUIT_ID
	ORDER BY TEMPO) AS M
INNER JOIN  DRIVERS D
ON D.DRIVER_ID = M.DRIVER_ID
ORDER BY TEMPO;




Pergunta: Pilotos que perderam mais tempo nos pit stops


Consulta:
SELECT D.DRIVER_ID, D.FORENAME, D.SURNAME, D.NUMBER, SEC_TO_TIME(FLOOR((SUM(P.MILISECONDS))/1000)) AS TEMPO_TOTAL, COUNT(D.DRIVER_ID) AS NUMERO_PARADAS
FROM PIT_STOPS P	
INNER JOIN DRIVERS D
ON D.DRIVER_ID = P.DRIVER_ID
GROUP BY D.DRIVER_ID
ORDER BY TEMPO_TOTAL DESC;
